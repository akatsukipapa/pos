function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); var rawSheet = ss.getSheetByName("シート1") || ss.getSheets()[0]; var passSheet = ss.getSheetByName("pass"); var correctPass = passSheet ? passSheet.getRange("A1").getValue().toString() : ""; // 履歴取得 (action: 'fetch') if (data.action === 'fetch') { if (String(data.password) !== correctPass) { return ContentService.createTextOutput(JSON.stringify({ error: "パスワードが違います" })).setMimeType(ContentService.MimeType.JSON); } return fetchHistory(ss, rawSheet, data.month); } // 売上保存 rawSheet.appendRow([data.timestamp, data.items, data.total]); updateMonthlySummary(ss, rawSheet); return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } catch (error) { return ContentService.createTextOutput("Error: " + error.toString()).setMimeType(ContentService.MimeType.TEXT); } } function fetchHistory(ss, rawSheet, targetMonth) { var allData = rawSheet.getDataRange().getValues(); var monthlySummary = {}; var details = []; for (var i = 1; i < allData.length; i++) { var dateVal = new Date(allData[i][0]); if (isNaN(dateVal)) continue; var monthKey = Utilities.formatDate(dateVal, "JST", "yyyy年MM月"); var amount = Number(allData[i][2]); monthlySummary[monthKey] = (monthlySummary[monthKey] || 0) + amount; if (targetMonth && monthKey === targetMonth) { details.push({ date: Utilities.formatDate(dateVal, "JST", "MM/dd HH:mm"), price: amount }); } } return ContentService.createTextOutput(JSON.stringify({ summary: monthlySummary, details: details })).setMimeType(ContentService.MimeType.JSON); } function updateMonthlySummary(ss, rawSheet) { var summarySheet = ss.getSheetByName("月別") || ss.insertSheet("月別"); summarySheet.clear().appendRow(["月", "合計金額"]); var allData = rawSheet.getDataRange().getValues(); var monthlyTotals = {}; for (var i = 1; i < allData.length; i++) { var dateVal = new Date(allData[i][0]); if (isNaN(dateVal)) continue; var monthKey = Utilities.formatDate(dateVal, "JST", "yyyy年MM月"); monthlyTotals[monthKey] = (monthlyTotals[monthKey] || 0) + Number(allData[i][2]); } var sortedMonths = Object.keys(monthlyTotals).sort().reverse(); var grandTotal = 0; for (var j = 0; j < sortedMonths.length; j++) { summarySheet.appendRow([sortedMonths[j], monthlyTotals[sortedMonths[j]]]); grandTotal += monthlyTotals[sortedMonths[j]]; } summarySheet.appendRow(["【年合計】", grandTotal]); }
